  \newpage
  \item 設 $f(x)$ 為一多項式，且 $x^2f(x)$ 除以 $x^2 + 2$ 的餘式為 $-20x - 24$。則 $f(x)$ 除以 $x^2 + 2$ 的餘式為 $\blank{8em}$。
\end{enumerate}
